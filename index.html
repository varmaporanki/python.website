Find Out How Many Days You've Lived!

from datetime import datetime from pyscript import Element def calculate_days(*args): birth_input = Element("birthdate").element.value birth_date = datetime.strptime(birth_input, "%Y-%m-%d") today = datetime.today() lived_days = (today - birth_date).days Element("output").element.innerText = f"You have lived for {lived_days} days!"
